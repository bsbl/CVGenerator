\newcommand{\experience}[9]{{

  }} % company

    \flushleft\small{\faCalendar \ #3 \faMapMarker \ #4}\flushleft
  }}
  \vspace{0ex}
  {{ % experience details
    #5
  }}
  % technos - can be empty:
  \ifthenelse{\isempty{#6}}%
    {} % no techno specified
    {
      \vspace{-0.1ex}
      \flushleft{\slshape\footnotesize\color{gray}
              \begin{addmargin}[1.8em]{0em}
                   \textbf{Technologies:\ }#6
              \end{addmargin}
       }
    }% show specified technos
}}
